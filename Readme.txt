using System;using System.Collections.Generic;using System.Linq;using System.Text;using System.Threading.Tasks;namespace ConsoleApplication2{	class Program	{		static void Main(string[] args)		{			int[] array1 = { 1, 3, 5, 7, 9, 11 };			int[] array2 = { 2, 4, 6, 8, 10, 12 };			int[] array3 = new int[12];			int k = 0;			int j = 0;			int i = 0;			for (k = 0; k < array3.Length; k++)			{				if (array1[i] < array2[j])				{					array3[k] = array1[i];					i = i + 1;				}				else				{					array3[k] = array2[j];					j = j + 1;								}			}						foreach (var item in array3)			{				Console.WriteLine(item);				}		}	}}
